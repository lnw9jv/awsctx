# awsctx Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use the appropriate execution skill (`executing-plans` or `subagent-driven-development`) to implement this plan.

**Goal:** Build a kubectx-style AWS profile switcher in Go that sets/unsets `AWS_PROFILE` via a shell function wrapper.
**Architecture:** A Go CLI (cobra) prints shell-evaluable output (`export`/`unset`); a shell wrapper function evals it so the env var propagates to the parent shell. Profile list is parsed from `~/.aws/config`. Previous profile is persisted to `~/.cache/awsctx/previous`.
**Tech Stack:** Go 1.22+, cobra, go-fzf, gopkg.in/ini.v1
**Assumptions:**
- Assumes `~/.aws/config` uses standard INI format with `[profile <name>]` sections — will NOT work with non-standard config layouts.
- Assumes shell wrapper is sourced by the user via `awsctx shell-init` — binary alone cannot mutate parent shell env.
- Assumes `AWS_CONFIG_FILE` env var is respected for custom config paths.

**Step 2: Implement `internal/aws/config.go`**

> **Key:** `ini.v1` always emits a synthetic root `"DEFAULT"` section (empty, no user-defined keys). The real `[default]` block also appears as `"DEFAULT"` but has keys. Detect the real default by checking `s.HasKey` on at least one known key, OR use `ini.LoadSources` with `IgnoreInlineComment`. The simplest reliable approach: iterate sections, skip the synthetic DEFAULT (which has 0 keys and is always the first section), and treat any subsequent `"DEFAULT"` section (which ini.v1 never produces — AWS config only ever has one default) as not applicable. Instead, use `ini.Load` with `InsensitiveSections: false` and rely on section ordering: the synthetic DEFAULT is always `cfg.Sections()[0]`; skip index 0 explicitly.

```go
package aws

import (
    "fmt"
    "os"
    "strings"
    "gopkg.in/ini.v1"
)

func ConfigPath() string {
    if p := os.Getenv("AWS_CONFIG_FILE"); p != "" {
        return p
    }
    home, _ := os.UserHomeDir()
    return home + "/.aws/config"
}

// LoadProfiles returns all profile names from the given config file path.
// [default] → "default", [profile foo] → "foo". Skips the synthetic ini root section.
func LoadProfiles(configPath string) ([]string, error) {
    cfg, err := ini.LoadSources(ini.LoadOptions{
        IgnoreInlineComment: true,
    }, configPath)
    if err != nil {
        return nil, fmt.Errorf("cannot read AWS config %s: %w", configPath, err)
    }
    var profiles []string
    for i, s := range cfg.Sections() {
        if i == 0 {
            // skip ini's synthetic root DEFAULT section (always index 0, always empty)
            continue
        }
        name := s.Name()
        if name == "DEFAULT" {
            // real [default] block — ini.v1 uppercases it
            profiles = append(profiles, "default")
            continue
        }
        if strings.HasPrefix(name, "profile ") {
            profiles = append(profiles, strings.TrimPrefix(name, "profile "))
        }
    }
    return profiles, nil
}
```

```go
package state

import (
    "errors"
    "os"
    "path/filepath"
    "strings"
)

var ErrNoPrevious = errors.New("no previous profile — switch profiles first")

type State struct{ dir string }

func New(dir string) *State { return &State{dir: dir} }

func DefaultDir() string {
    home, _ := os.UserHomeDir()
    return filepath.Join(home, ".cache", "awsctx")
}

func (s *State) previousPath() string {
    return filepath.Join(s.dir, "previous")
}

func (s *State) SetPrevious(profile string) error {
    if err := os.MkdirAll(s.dir, 0o700); err != nil {
        return err
    }
    return os.WriteFile(s.previousPath(), []byte(profile), 0o600)
}

func (s *State) GetPrevious() (string, error) {
    b, err := os.ReadFile(s.previousPath())
    if os.IsNotExist(err) {
        return "", ErrNoPrevious
    }
    if err != nil {
        return "", err
    }
    return strings.TrimSpace(string(b)), nil
}
```

```go
package cmd

import (
    "errors"
    "fmt"
    "os"

    awscfg "github.com/lnw9jv/awsctx/internal/aws"
    "github.com/lnw9jv/awsctx/internal/state"

func switchProfile(profile string) error {
    st := state.New(state.DefaultDir())
    profiles, err := awscfg.LoadProfiles(awscfg.ConfigPath())
    if err != nil {
        return err
    }
    found := false
    for _, p := range profiles {
        if p == profile {
            found = true
            break
        }
    }
    if !found {
        return fmt.Errorf("profile %q not found in %s", profile, awscfg.ConfigPath())
    }

    current := os.Getenv("AWS_PROFILE")
    if current != "" && current != profile {
        _ = st.SetPrevious(current)
    }

    fmt.Printf("export AWS_PROFILE=%s\n", profile)
    return nil
}
```

### Task 7: Switch to previous profile (`awsctx -`)

**Files:**

**Does NOT cover:** `awsctx --previous` long form (not in spec).

**Step 1: Handle `-` as special argument in `RunE`**

`-` is not a valid flag so cobra passes it as an arg.

> **Critical:** `switchProfile` only saves previous when `current != profile`. When using `awsctx -`, we must explicitly save the current profile as previous *before* calling `switchProfile`, so two consecutive `awsctx -` calls correctly toggle between two profiles.

Update the args handler:
```go
if len(args) == 1 && args[0] == "-" {
    st := state.New(state.DefaultDir())
    prev, err := st.GetPrevious()
    if err != nil {
        return err
    }
    // save current before switching so awsctx - is idempotently toggleable
    if current := os.Getenv("AWS_PROFILE"); current != "" {
        _ = st.SetPrevious(current)
    }
    fmt.Printf("export AWS_PROFILE=%s\n", prev)
    return nil
}
```

Note: the `-` branch prints the export directly and sets previous itself, bypassing `switchProfile`'s validation. This is intentional — the previous profile was already validated when it was first switched to.

```go
package picker

import (
    "fmt"
    fzf "github.com/ktr0731/go-fzf"
)

// Pick opens an interactive fuzzy finder and returns the selected item.
// currentProfile is highlighted with a checkmark if it appears in items.
func Pick(items []string, currentProfile string) (string, error) {
    if len(items) == 0 {
        return "", fmt.Errorf("no AWS profiles found")
    }

    f, err := fzf.New(fzf.WithPrompt("AWS Profile> "))
    if err != nil {
        return "", err
    }

    idxs, err := f.Find(items, func(i int) string {
        if items[i] == currentProfile {
            return items[i] + " ✓"
        }
        return items[i]
    })
    if err != nil {
        return "", err
    }
    if len(idxs) == 0 {
        return "", fmt.Errorf("no profile selected")
    }
    return items[idxs[0]], nil
}
```

```go
package cmd

import (
    "fmt"
    "github.com/spf13/cobra"
)

var shellInitCmd = &cobra.Command{
    Use:   "shell-init [zsh|bash|fish]",
    Short: "Print shell integration snippet",
    Args:  cobra.ExactArgs(1),
    RunE: func(cmd *cobra.Command, args []string) error {
        switch args[0] {
        case "zsh", "bash":
            fmt.Print(shellWrapperPosix)
        case "fish":
            fmt.Print(shellWrapperFish)
        default:
            return fmt.Errorf("unsupported shell %q — use zsh, bash, or fish", args[0])
        }
        return nil
    },
}

const shellWrapperPosix = `
# awsctx shell integration
awsctx() {
  local out
  out=$(command awsctx "$@") || return $?
  if [[ "$out" == export* || "$out" == unset* ]]; then
    eval "$out"
  else
    echo "$out"
  fi
}
`

const shellWrapperFish = `
# awsctx shell integration
function awsctx
  set out (command awsctx $argv)
  or return $status
  if string match -q 'export *' $out; or string match -q 'unset *' $out
    eval (echo $out | sed 's/export /set -gx /' | sed 's/unset /set -e /')
  else
    echo $out
  end
end
`

func init() {
    rootCmd.AddCommand(shellInitCmd)
}
```

```go
package cmd

import (
    "github.com/spf13/cobra"
    awscfg "github.com/lnw9jv/awsctx/internal/aws"
)

var completionCmd = &cobra.Command{
    Use:       "completion [zsh|bash|fish]",
    Short:     "Generate shell completion script",
    Args:      cobra.ExactArgs(1),
    ValidArgs: []string{"zsh", "bash", "fish"},
    RunE: func(cmd *cobra.Command, args []string) error {
        switch args[0] {
        case "zsh":
            return rootCmd.GenZshCompletion(cmd.OutOrStdout())
        case "bash":
            return rootCmd.GenBashCompletion(cmd.OutOrStdout())
        case "fish":
            return rootCmd.GenFishCompletion(cmd.OutOrStdout(), true)
        }
        return nil
    },
}

func init() {
    rootCmd.AddCommand(completionCmd)

    // Dynamic completion: profile names as positional arg completions
    rootCmd.ValidArgsFunction = func(cmd *cobra.Command, args []string, toComplete string) ([]string, cobra.ShellCompDirective) {
        if len(args) != 0 {
            return nil, cobra.ShellCompDirectiveNoFileComp
        }
        profiles, err := awscfg.LoadProfiles(awscfg.ConfigPath())
        if err != nil {
            return nil, cobra.ShellCompDirectiveError
        }
        return profiles, cobra.ShellCompDirectiveNoFileComp
    }
}
```
